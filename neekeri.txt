local pastebinUrls = {
    "https://pastebin.com/raw/fm7cbpby",
    "https://pastebin.com/raw/51rw40CS", 
    "https://pastebin.com/raw/gzquQjTK"
}

local groupAudioIds = {}

local function loadAudioIdsFromPastebin(url)
    local success, result = pcall(function()
        return game:HttpGet(url)
    end)
    
    if success then
        for audioId in result:gmatch("[^\r\n]+") do
            table.insert(groupAudioIds, audioId)
        end
    else
        warn("Failed to fetch audio IDs from " .. url .. ": " .. tostring(result))
    end
end

for _, url in ipairs(pastebinUrls) do
    loadAudioIdsFromPastebin(url)
end

local Rayfield = loadstring(game:HttpGet('https://sirius.menu/rayfield'))()
local Window = Rayfield:CreateWindow({
    Name = "Rontto's Boombox",
    LoadingTitle = "Loading Rontto's Boombox",
    LoadingSubtitle = "by Ronttosaurus54321",
    ConfigurationSaving = {
        Enabled = true,
        FolderName = "NEWWORLDBOOMBOX",
        FileName = "RonttoBoomboxConfig"
    }
})

local MainTab = Window:CreateTab("Audio Player", nil)
local Section = MainTab:CreateSection("Select and Play Audio")

-- Sound object to play selected audios
local sound = Instance.new("Sound", game.Workspace)
sound.Name = "DynamicAudio"
sound.Looped = true
sound.Volume = 1

local GroupAudioDropdown = MainTab:CreateDropdown({
    Name = "Select Group Audio",
    Options = groupAudioIds, -- Use groupAudioIds as the options
    Callback = function(selectedAudioId)
        if selectedAudioId then
            sound.SoundId = "rbxassetid://" .. selectedAudioId
            print("Selected Group Audio ID: " .. selectedAudioId)
            sound:Play()
        else
            warn("Invalid Audio ID selected or conversion failed.")
        end
    end
})

MainTab:CreateButton({
    Name = "Play Sound",
    Callback = function()
        if not sound.IsPlaying then
            sound:Play()
        else
            sound:Resume()
        end
    end
})

MainTab:CreateButton({
    Name = "Pause Sound",
    Callback = function()
        sound:Pause()
    end
})

MainTab:CreateButton({
    Name = "Stop Sound",
    Callback = function()
        sound:Stop()
    end
})
